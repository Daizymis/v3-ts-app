interface CardProps {
  tag?: string; //图片上标签
  num?: number; //商品数量
  desc?: string; //商品描述

  title?: string; // 标题

  centered?: boolean;

  originPrice?: typeof numericProp; //
  tags?: Array<string>; // 描述下面的标签
}

##### thumb 自定义图片

##### bottom 自定义下方区域
